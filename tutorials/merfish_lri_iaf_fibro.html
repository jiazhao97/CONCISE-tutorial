# --- graph ---
nodes <- tibble::tibble(
  name = c("IAF", f_celltypes)
)
edges <- tibble::tibble(
  from = iaf_celltypes,
  to   = f_celltypes,
  strength = LR_npairs
)
g <- tbl_graph(nodes = nodes, edges = edges, directed = TRUE)
g <- g %>%
  activate(edges) %>%
  mutate(receiver = .N()$name[to])   # receiver = target node name

# --- layout ---
layout <- create_layout(g, layout = "linear", circular = TRUE)

# basics
rad_out <- 1.17   # how far to push most labels radially
delta_y <- 0.10   # vertical nudge for extreme left/right nodes
# identify nodes near pure left/right (within ~15 degrees)
theta <- atan2(layout$y, layout$x)
near_lr <- abs(cos(theta)) > cos(pi/12)  # TRUE near 0° or 180°

# default: push labels radially outward, keep horizontal
layout$lx <- layout$x * rad_out
layout$ly <- layout$y * rad_out
layout$vjust_lab <- 0.5  # centered by default

# override for far-left/right nodes: put label above/below instead
layout$lx[near_lr] <- layout$x[near_lr]                      # keep x ~ node
layout$ly[near_lr] <- layout$y[near_lr] + ifelse(layout$y[near_lr] >= 0,
                                                 +delta_y, -delta_y)
layout$vjust_lab[near_lr] <- ifelse(layout$y[near_lr] >= 0, 0, 1)  # above/below

pad <- max(abs(c(layout$lx, layout$ly))) * 1

# --- plot ---
p <- ggraph(layout) +
  geom_edge_arc(
    aes(width = strength, alpha = strength, color = receiver),
    arrow   = arrow(type = "closed", length = unit(3, "mm")),
    end_cap = circle(2.2, "mm"),
    lineend = "round"
  ) +
  scale_edge_width(range = c(0.4, 3.5), guide = "none") +
  scale_edge_alpha(range = c(0.25, 1), guide = "none") +
  scale_edge_color_manual(
    values = c(
      "Fibroblast 1" = "#4B371C",
      "Fibroblast 2" = "#80471C"
    )
  ) +
  geom_node_point(size = 3.5, color = "#222222") +
  
  # labels: horizontal, away from nodes; left/right become above/below
  geom_node_text(
    # aes(x = lx, y = ly, label = name, color = name),
    aes(x = lx, y = ly, label = name),
    angle = 0,
    hjust = 0.5,
    vjust = layout$vjust_lab,
    size  = 4
  ) +
  
  theme_void() +
  coord_equal(xlim = c(-pad, pad), ylim = c(-pad, pad), clip = "off")
