Spatial ligand-receptor interaction inference with CONCISE

#### Run CONCISE for inferring spatial interactions

pairdb <- CONCISE(
  pairdb,
  rawcount,
  loc,
  Wx_distance_threshold = 1.5)
## Creating spatial kernels...
##  marginal kernel 1
##  marginal kernel 2
##  marginal kernel 3
##  interaction kernel
## Done!
## Iterating over choices of kernels and estimating parameters of marginal distributions...
##  marginal kernel 1
##  marginal kernel 2
##  marginal kernel 3
## Done!
## Estimating parameters of the joint spatial process and performing statistical inference...
##  pre-calculation for parameter estimation
##  parameter estimation for each pair in pairdb
##  pre-calculation for statistical inference
##  terms not related to marginal kernels
##  terms related to only one marginal kernel
##  terms related to both two marginal kernels
##  statistical inference for each pair in pairdb
## Done!
P_value <- pairdb$p_value
P_value_adjust <- p.adjust(P_value, method = "BY")
pairdb$p_value_adjust <- P_value_adjust
res_sig_BY <- pairdb[pairdb$p_value_adjust < 0.05, ]
print(paste0("CONCISE reported ", dim(res_sig_BY)[1], " ligand-receptor interaction pairs."))
## [1] "CONCISE reported 43 ligand-receptor interaction pairs."
