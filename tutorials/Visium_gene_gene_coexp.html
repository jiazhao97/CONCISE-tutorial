#### p-value threshold based

pval_list <- c(0.0001, 0.001, 0.01, 0.02, 0.05)
overlap_pval_list <- c()

for (pval in pval_list) {
  corrdb <- corrdb_raw
  n_hvg <- length(unique(corrdb$gene1))
  hvg_list <- corrdb$gene2[1:n_hvg]
  corrdb$p_adjust <- p.adjust(as.vector(corrdb$p_value), method = "BY")
  corrdb <- corrdb[corrdb$gene1 != corrdb$gene2, ]
  corrdb.sig <- corrdb[corrdb$p_adjust < pval, ]

  corrdb.sig$gene1_id <- gene_name2id_list[corrdb.sig$gene1]
  corrdb.sig$gene2_id <- gene_name2id_list[corrdb.sig$gene2]
  corrdb.sig <- corrdb.sig[!is.na(corrdb.sig$gene1_id), ]
  corrdb.sig <- corrdb.sig[!is.na(corrdb.sig$gene2_id), ]
  corrdb.sig$gene_id_pair <- paste(corrdb.sig$gene1_id, corrdb.sig$gene2_id, sep = "_")
  corrdb.sig$gene_name_pair <- paste(corrdb.sig$gene1, corrdb.sig$gene2, sep = "_")

  overlap_pval_list <- c(overlap_pval_list, length(intersect(corrdb.sig$gene_id_pair, string.db$pair)) / (dim(corrdb.sig)[1] / 2))

  print(paste0("p-value threshold = ", pval, ", STRING overlap proportion = ", length(intersect(corrdb.sig$gene_id_pair, string.db$pair)) / (dim(corrdb.sig)[1] / 2)))
}
